        if element is None:

        if element is not None:
            value = element.text
            data.append(value)
        else:
            data.append(None)

    data.append(file_path)

    # Create the DataFrame inside the function
    df = pd.DataFrame([data], columns=list(all_tags) + ["pathToXmlFile"])
    path_to_xml_file_split = df["pathToXmlFile"].str.split("/")
    return df
