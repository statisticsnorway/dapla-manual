Denne oversikten er ment som et oppslagsverk for Dapla-relaterte fellessystemer, tjenester og verktøy i SSB. Den skal gi en kort og overordnet forklaring av navn som ofte brukes i arbeid med statistikkproduksjon, forskning og forvaltning av data.

Sist oppdatert: 23. juni 2026

systemer = FileAttachment("fellessystemer_dapla_hovedoversikt.csv").csv({typed: false})
notBlank = (x) => x !== undefined && x !== null && String(x).trim() !== ""
normaliser = (x) => String(x ?? "")
  .toLowerCase()
  .normalize("NFD")
  .replace(/[\u0300-\u036f]/g, "")

sorterNorsk = (a, b) => String(a).localeCompare(String(b), "nb-NO")

kategorier = ["Alle", ...Array.from(new Set(systemer.map(d => d.Kategori).filter(notBlank))).sort(sorterNorsk)]
typer = ["Alle", ...Array.from(new Set(systemer.map(d => d.Type).filter(notBlank))).sort(sorterNorsk)]
viewof filtre = Inputs.form({
  kategori: Inputs.select(kategorier, {label: "Kategori"}),
  type: Inputs.select(typer, {label: "Type"}),
  sok: Inputs.text({label: "Søk", placeholder: "Søk etter navn, målgruppe eller forklaring"})
})
filtrerteSystemer = systemer.filter(d => {
  const riktigKategori = filtre.kategori === "Alle" || d.Kategori === filtre.kategori
  const riktigType = filtre.type === "Alle" || d.Type === filtre.type
  const q = normaliser(filtre.sok)
  const sokefelt = [d.Navn, d["Målgruppe"], d.Forklaring, d.Type, d.Kategori].map(normaliser).join(" ")
  return riktigKategori && riktigType && (q === "" || sokefelt.includes(q))
})
html`<div class="system-results-summary">
  <strong>${filtrerteSystemer.length}</strong> av <strong>${systemer.length}</strong> oppføringer vises.
</div>

<div class="system-card-list">
${filtrerteSystemer.map(d => {
  const tags = [d.Type, d.Kategori].filter(notBlank)
  const kilde = d["Kilde / mer informasjon"]
  return html`<article class="system-card">
    <h3>${d.Navn}</h3>
    ${tags.length ? html`<div class="system-tags">${tags.map(t => html`<span class="system-tag">${t}</span>`)}</div>` : ""}
    ${notBlank(d.Forklaring) ? html`<p>${d.Forklaring}</p>` : ""}
    ${notBlank(d["Målgruppe"]) ? html`<p class="system-meta"><strong>Målgruppe:</strong> ${d["Målgruppe"]}</p>` : ""}
    ${notBlank(kilde) ? html`<a class="system-link" href=${kilde} target="_blank" rel="noopener noreferrer">Mer informasjon</a>` : ""}
  </article>`
})}
</div>`
